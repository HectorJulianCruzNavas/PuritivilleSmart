import random

def predict_traffic():
    # Simulación de predicción de tráfico basada en datos históricos (aquí datos aleatorios)
    traffic_conditions = ["Despejado", "Moderado", "Congestionado"]
    prediction = random.choice(traffic_conditions)
    
    # Aquí se añadiría el código de machine learning para predecir tráfico real
    return prediction
